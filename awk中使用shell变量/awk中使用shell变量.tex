\section{awk中使用shell变量}
	例子如下：
	\begin{lstlisting}
	#!/bin/bash

	MAC=`awk '/virbr0/ { print $4 }' /proc/net/arp`
	domains=`virsh list | awk '/running/ {print $2}'`

	for VM in $domains
	do
		line=`virsh dumpxml $VM | grep "mac address"`
		address=`echo $line | awk -F "" '
		{
			for(i=1; i<=NF; i++)
			{
				if($i ~ /'\''/)
					break
			}
			i++
			str=""
			for(;i<=NF;i++)
			{
				if($i ~ /'\''/)
					break
				temp=$i
				str=(str temp)
			}
			print str
		}'`

		# 使用两个单引号包围shell变量，就可以使用shell变量
		ip=`awk '/'$address'/ {print $1}' /proc/net/arp`
		echo $ip
	done
	\end{lstlisting}
